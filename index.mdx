export function StatusTable({ status }) {
  const [col, setCol] = useState(null)
  const [dir, setDir] = useState('asc')

  const rows = allRows.filter(r => r.status === status)

  const toggle = (c) => {
    if (col === c) setDir(d => d === 'asc' ? 'desc' : 'asc')
    else { setCol(c); setDir('asc') }
  }

  const arrow = (c) => col === c ? (dir === 'asc' ? ' ↑' : ' ↓') : ' ↕'

  const sorted = col === null ? rows : [...rows].sort((a, b) => {
    const map = { id: 'id', desc: 'desc', cat: 'cat', area: 'area', prio: 'priord', data: 'data' }
    const key = map[col]
    const av = a[key]; const bv = b[key]
    const cmp = typeof av === 'number' ? av - bv : String(av).localeCompare(String(bv), 'pt')
    return dir === 'asc' ? cmp : -cmp
  })

  const thStyle = { cursor: 'pointer', userSelect: 'none', whiteSpace: 'nowrap' }

  if (rows.length === 0) return (
    <table><thead><tr>
      <th>ID</th><th>Descrição</th><th>Categoria</th><th>Área</th><th>Prioridade</th><th>Data</th>
    </tr></thead><tbody><tr><td>—</td><td>—</td><td>—</td><td>—</td><td>—</td><td>—</td></tr></tbody></table>
  )

  return (
    <table>
      <thead>
        <tr>
          <th style={thStyle} onClick={() => toggle('id')}>ID{arrow('id')}</th>
          <th style={thStyle} onClick={() => toggle('desc')}>Descrição{arrow('desc')}</th>
          <th style={thStyle} onClick={() => toggle('cat')}>Categoria{arrow('cat')}</th>
          <th style={thStyle} onClick={() => toggle('area')}>Área{arrow('area')}</th>
          <th style={thStyle} onClick={() => toggle('prio')}>Prioridade{arrow('prio')}</th>
          <th style={thStyle} onClick={() => toggle('data')}>Data{arrow('data')}</th>
        </tr>
      </thead>
      <tbody>
        {sorted.map(r => (
          <tr key={r.id}>
            <td><a href={r.href}>{r.id}</a></td>
            <td>{r.desc}</td>
            <td>{r.cat}</td>
            <td>{r.area}</td>
            <td>{r.prio}</td>
            <td>{r.data}</td>
          </tr>
        ))}
      </tbody>
    </table>
  )
}
